# GridSVG: compute wire segments once (reused for black + white)
seg_full  <- thin(dat, 40)
seg_y2    <- thin(dat |> filter(row_number() <= which.max(y2 > y)), 25)
seg_y3    <- thin(dat |> filter(row_number() >= which.max(y3 < y2)), 25)
seg_ny3   <- thin(dat |> filter(row_number() >= last(which(-y3 < -y))), 25)
seg_ny2   <- thin(dat |> filter(row_number() <= last(which(-y2 > -y3))), 25)

wire_segs <- list(
  list(x = seg_full$x,  y =  seg_full$y),
  list(x = seg_full$x,  y = -seg_full$y),
  list(x = seg_y2$x,    y =  seg_y2$y2),
  list(x = seg_y3$x,    y =  seg_y3$y3),
  list(x = seg_ny3$x,   y = -seg_ny3$y3),
  list(x = seg_ny2$x,   y = -seg_ny2$y2)
)

draw_wire_spline <- function(col_wire, filename) {
  pdf(NULL, width = 3, height = 2)
  grid.newpage()
  pushViewport(vp_logo)
  for (s in wire_segs) {
    grid.xspline(unit(s$x, "native"), unit(s$y, "native"),
                 shape = 1, open = TRUE,
                 gp = gpar(col = col_wire, lwd = 1.5, fill = NA))
  }
  popViewport()
  grid.export(filename)
  dev.off()
}
